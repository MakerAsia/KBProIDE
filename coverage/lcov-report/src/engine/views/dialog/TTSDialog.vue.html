<template>
    <v-card>
        <v-card-title>
            <span class="headline">Text To Speech</span>
        </v-card-title>
        <v-divider></v-divider>
        <v-card-text>
            <v-container>
                <v-layout column>
                    <p class="subheading">
                        Type the words (support only listed auto complete)
                    </p>
                    <vue-tags-input
                            v-model="tag"
                            :tags="tags"
                            :allow-edit-tags="false"
                            :autocomplete-filter-duplicates="false"
                            :add-from-paste="false"
                            :add-only-from-autocomplete="true"
                            :avoid-adding-duplicates="false"
                            @tags-changed="newTags => tags = newTags"
                            :autocomplete-items="filteredItems"
                    />
                </v-layout>
            </v-container>
        </v-card-text>
        <v-card-actions>
            <v-spacer></v-spacer>
            <v-btn color="blue darken-1" flat @click="close">Cancel</v-btn>
            <v-btn color="blue darken-1" flat @click="OK">OK</v-btn>
        </v-card-actions>
    </v-card>
</template>
<script>
  import VueTagsInput from "@johmun/vue-tags-input";
  import Words from "./tts";
  export default {
    components: {
      VueTagsInput,
    },
    /*props:{
      tags: {
        type: Array,
        required: false
      },
    },*/
    data() {
      return {
        autocompleteItems: Words,
        tag: "",
        tags : []
      };
    },
    computed:{
      filteredItems() {
        return this.autocompleteItems.filter(i => {
          return i.text.toLowerCase().startsWith(this.tag.toLocaleLowerCase());
        }).sort((a,b)=>{
          return a.text.length - b.text.length;
        });
      },
    },
    methods : {
      close(){
        this.$emit('close');
      },
      OK(){
        this.$emit('result',this.tags.map(obj => obj.text));
      }
    }
  };
</script>
