<template>
    <div class="key"
         v-bind:class="isMajorOrSilence"
         @click="sound(tone)">
    </div>
</template>
 
<script>
  export default {
    name: "piano-key",
    props: {
      tone: {
        type: String,
        required: true,
      },
    },
    data() {
      return {};
    },
    computed: {
      isMajorOrSilence: function() {
        return {
          major: this.tone.includes("#"),
          silence: this.tone === "SIL",
        };
      },
    },
    methods: {
      sound: function(tone) {
        this.$parent.pressedNote(tone);
      },
    },
  };
</script>
